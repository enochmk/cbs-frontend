            <com:CommandId>ActiveFirst</com:CommandId>

            <com:SequenceId>1</com:SequenceId>
            <com:RequestType>Event</com:RequestType>
            <com:SessionEntity>
               <com:Name></com:Name>
               <com:Password></com:Password>
               <com:RemoteAddress></com:RemoteAddress>
            </com:SessionEntity>
